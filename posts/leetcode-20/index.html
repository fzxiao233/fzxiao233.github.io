func isValid(s string) bool {
    if len(s)%2 != 0 {
        return false
    }
    pairs := map[byte]byte{
        '}':'{',
        ']':'[',
        ')':'('}
    var stack []byte
    for i:=0; i<len(s); i++{
        _, ok := pairs[s[i]]
        if ok {
            if len(stack) == 0 || stack[len(stack)-1] != pairs[s[i]] {
                return false
            }
            stack = stack[:len(stack)-1]
        } else {
            stack = append(stack, s[i])
        }
    }
    return len(stack) == 0
}

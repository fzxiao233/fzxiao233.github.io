class Solution {
    public int countPrimes(int n) {
        int count = 0;
        for(int i = 2; i < n; i++) {
            int t = 0;
            for(int c = 2; c < i; c++) {
                if(i % c == 0) {
                    t++;
                }
            }
            if(t == 0) {
                count++;
            }
        }
        return count;
    }
}

class Solution {
    public int countPrimes(int n) {
        boolean[] result = new boolean[n];
        Arrays.fill(result, false);
        for(int i = 2; i * i < n; i++) {
            if(result[i] == false) {
                for(int j = i * i; j < n; j = j + i) {
                    result[j] = true;
                }
            }
        }
        int count = 0;
        for (int i = 2; i < n; i++) {
            if(result[i] == false) {
                count++;
            }
        }
        return count;
    }
}
